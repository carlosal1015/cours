\begin{enumerate}
\item
Soit $x∈Λ$. On a
  \begin{align*}
    x∈f^{-1}(Ω∖Y)
    &\iff f(x)∈Ω∖Y \\
    &\iff f(x)∉Y \\
    &\iff x∉f^{-1}(Y) \\
    &\iff x∈Λ∖f^{-1}(Y).
  \end{align*}

\item
Soit $x∈Λ$. On a
  \begin{align*}
    x∈f^{-1} \Pa{ ⋃_{i∈I} X_i }
    &\iff f(x)∈⋃_{i∈I} X_i \\
    &\iff ∃i∈I \+ f(x)∈X_i \\
    &\iff ∃i∈I \+ x∈f^{-1}(X_i) \\
    &\iff x∈⋃_{i∈I}f^{-1}(X_i).
  \end{align*}

\item
Soit $x∈Λ$. On a
  \begin{align*}
    x∈f^{-1} \Pa{ ⋂_{i∈I} X_i }
    &\iff f(x)∈⋂_{i∈I} X_i \\
    &\iff ∀i∈I \+ f(x)∈X_i \\
    &\iff ∀i∈I \+ x∈f^{-1}(X_i) \\
    &\iff x∈⋂_{i∈I}f^{-1}(X_i).
  \end{align*}

\end{enumerate}
